                def capture(args={})
                        filter = args[:filter]
                        start = args[:start] || true
                        if start
                                begin
                                        @stream = Pcap.open_live(@iface,@snaplen,@promisc,@timeout)
                                rescue RuntimeError
                                        $stderr.print "Are you root? Error: "
                                        raise
                                end
                                bpf(:filter=>filter) if filter
                        else
                                @stream = []
                        end
                        @stream
                end
