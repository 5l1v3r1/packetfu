                def self.translate(arr)
                        translated_opts = arr.collect do |opt|
                                case opt
                                when "\x00"; "EOL"
                                when "\x01"; "NOP"
                                else
                                        if opt[2].class == String
                                                TcpOpts.option_to_s(opt[0].unpack("C")[0],opt[2])
                                        else
                                                "INVALID:#{opt.pack("aC")}"
                                        end
                                end
                        end
                        translated_opts.join(",")
                end
