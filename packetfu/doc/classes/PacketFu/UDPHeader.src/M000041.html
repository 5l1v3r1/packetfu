                def udp_recalc(args=:all)
                        case args
                        when :udp_len
                                self.udp_len = udp_calc_len
                        when :all
                                self.udp_recalc(:udp_len)
                        else
                                raise ArgumentError, "No such field `#{arg}'"
                        end
                end
