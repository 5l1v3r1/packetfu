                def icmp_recalc(arg=:all)
                        case arg.intern
                        when :icmp_sum
                                self.icmp_sum=icmp_calc_sum
                        when :all
                                self.icmp_sum=icmp_calc_sum
                        else
                                raise ArgumentError, "No such field `#{arg}'"
                        end
                end
