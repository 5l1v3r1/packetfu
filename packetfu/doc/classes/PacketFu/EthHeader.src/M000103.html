                def self.mac2str(mac)
                        if mac.split(/[:\x2d\x2e\x5f]/).size == 6
                                ret =       mac.split(/[:\x2d\x2e\x20\x5f]/).collect {|x| x.to_i(16)}.pack("C6")
                        else
                                raise ArgumentError, "Unkown format for mac address."
                        end
                        return ret
                end
