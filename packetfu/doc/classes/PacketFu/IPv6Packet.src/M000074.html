                def peek(args={})
                        peek_data = ["6 "]
                        peek_data << "%-5d" % self.to_s.size
                        peek_data << self.ipv6_saddr
                        peek_data << "->"
                        peek_data << self.ipv6_daddr
                        peek_data << " N:"
                        peek_data << self.ipv6_next.to_s(16)
                        peek_data.join
                end
