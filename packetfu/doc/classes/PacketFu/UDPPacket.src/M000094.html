                def peek(args={})
                        peek_data = ["U "]
                        peek_data << "%-5d" % self.to_s.size
                        peek_data << "%-21s" % "#{self.ip_saddr}:#{self.udp_src}"
                        peek_data << "->"
                        peek_data << "%21s" % "#{self.ip_daddr}:#{self.udp_dst}"
                        peek_data << "%23s" % "I:"
                        peek_data << "%04x" % self.ip_id
                        peek_data.join
                end
