                def udp_recalc(args=:all)
                        case args
                        when :udp_len
                                @udp_header.udp_recalc
                        when :udp_sum
                                udp_calc_sum
                        when :all
                                @udp_header.udp_recalc
                                udp_calc_sum
                        else
                                raise ArgumentError, "No such field `#{arg}'"
                        end
                end
